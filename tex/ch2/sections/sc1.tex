\section{Zweiseitige Z-Transformation und Konvergenzbereich}
	Gegeben: Signal oder Impulsantwort
	\begin{equation}
		x(n) \in \R \text{ od. } \C \qquad, n \in \Z
	\end{equation}
	\begin{align}
	X(z) &= Z\left( x(n) \right) = \sum\limits_{n = - \infty}^\infty	 x(n) z^{-n} \qquad, z = r \cdot e^{j\omega} \quad \in \C \\
	&= \dots + x(-1)z + x(0) + x(1)z^{-1} + \dots \nonumber
	\end{align}
